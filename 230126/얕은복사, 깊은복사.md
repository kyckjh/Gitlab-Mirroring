### 얕은복사

a = [a, b, [c, d]

b = a

b[0] = 1

###### # a : [1, b, [c, d]], b : [1, b, [c, d]]

###### # b는 a의 주소를 복사

### 깊은복사 + 얕은복사

a = [a, b, [c, d]

b = a[:]

b[0] = 1

b[2][0] = 1

###### # a : [a, b, [1, d]], b : [1, b, [1, d]]

###### # b는 a의 값을 복사했으나 b[2]에서는 원소가 배열이므로 주소를 복사

### 깊은복사

import copy

a = [a, b, [c, d]

b = copy.deepcopy(a)

b[2][0] = 1

###### # a : [a, b, [c, d]], b : [a, b, [1, d]]
